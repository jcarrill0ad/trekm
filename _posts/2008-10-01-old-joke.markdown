<code>
<pre>
class Economy
  def slowing?
   self.value.today &lt; self.value.yesterday
  end

  def slow_by(amount)
    self.value -= amount
  end
end

class Money
  def self.print(amount = 10000)
    0.upto(size) do
     self.economy &lt;&lt; Money.new
    end
    self.economy.slow_by(amount)
  end
end

economy = Economy.new('United States', 'United States Dollars')

while(economy.slowing?)
  Money.print
end
